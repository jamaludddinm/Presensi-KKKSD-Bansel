<script>

const API_URL = 'https://script.google.com/a/admin.sd.belajar.id/macros/s/AKfycbzBlGWQg8JnbTh0wfWT-OewHl1Lj9xYsFy1HMISN5kvH5f7p8es/exec';


let scanner = null;

let busy = false;

let sukses = 0;


function jsonp(url, cb) {

  const name = 'cb_' + Date.now() + '_' + Math.floor(Math.random() * 10000);

  window[name] = function (data) {

    cb(data);

    delete window[name];

    script.remove();

  };

  const script = document.createElement('script');

  script.src = url + (url.includes('?') ? '&' : '?') + 'callback=' + name;

  script.onerror = function () { cb(null); delete window[name]; script.remove(); };

  document.body.appendChild(script);

}


function setResult(text, cls) {

  const el = document.getElementById('result');

  el.textContent = text;

  el.className = cls;

}


if (API_URL.indexOf('PASTE_URL_WEB_APP_DISINI') !== -1) {

  document.getElementById('setupBox').style.display = 'block';

  setResult('Konfigurasi API_URL belum diisi.', 'err');

} else {

  jsonp(API_URL + '?action=getKegiatan', function (list) {

    const sel = document.getElementById('kegiatan');

    if (!list) {

      setResult('Gagal terhubung ke server. Cek API_URL / koneksi internet.', 'err');

      return;

    }

    sel.innerHTML = list.length

      ? '<option value="">-- pilih kegiatan --</option>'

      : '<option value="">Tidak ada kegiatan aktif</option>';

    list.forEach(function (k) {

      const o = document.createElement('option');

      o.value = k.id;

      o.textContent = k.nama + ' (' + k.tanggal + ')';

      sel.appendChild(o);

    });

    setResult('Pilih kegiatan, lalu arahkan kamera ke QR peserta', 'info');

  });


  document.getElementById('kegiatan').addEventListener('change', function () {

    if (this.value && !scanner) startScanner();

  });

}


function startScanner() {

  scanner = new Html5Qrcode('reader');

  scanner

    .start({ facingMode: 'environment' }, { fps: 10, qrbox: 220 }, onScanSuccess)

    .catch(function (err) {

      setResult('Gagal buka kamera: ' + err, 'err');

    });

}


function onScanSuccess(decodedText) {

  if (busy) return;

  const idKegiatan = document.getElementById('kegiatan').value;

  if (!idKegiatan) return;

  busy = true;

  setResult('Memproses...', 'info');


  const url = API_URL + '?action=scan&idKegiatan=' + encodeURIComponent(idKegiatan) +

    '&idPeserta=' + encodeURIComponent(decodedText);


  jsonp(url, function (r) {

    if (!r) {

      setResult('Gagal menghubungi server.', 'err');

    } else {

      setResult((r.success ? 'OK: ' : 'Gagal: ') + r.message, r.success ? 'ok' : 'err');

      if (r.success) {

        sukses++;

        document.getElementById('cnt').textContent = sukses;

      }

    }

    if (navigator.vibrate) navigator.vibrate(r && r.success ? 150 : [100, 50, 100]);

    setTimeout(function () { busy = false; }, 1800);

  });

}
